defined_text = {
	name = GetRepelInterLopersDesc
	random = no

	text = {
		localisation_key = sky_domain.2.d.yemo
		trigger = {
			has_country_flag = S50_upgraded_repel_interlopers
			any_neighbor_country = {
				NOT = { religion = sky_domain }
				NOT = { culture_group = mengi }
				NOT = { alliance_with = ROOT }
				any_owned_province = {
					region = shadow_swamp_region
					NOT = { is_core = ROOT }
				}
			}
		}
	}

	text = {
		localisation_key = sky_domain.2.d.default
		trigger = {
			always = yes
		}
	}
}
