
######################################################################
# Traitor-mages, who want to coup and institute magocracy
######################################################################
mage_disaster_rebels = {

	color = { 179 75 75 }

	area = nation 					# Stays within the nation
	government = theocracy			# Likes theocracy
	defection = none				# Does not defect
	independence = none				# Does not strive for independence
	defect_delay = 120
	unit_transfer = no
	gfx_type = culture_owner	
	will_relocate = yes	
	
	resilient = yes
	reinforcing = yes
	general = yes
	smart = yes
	
	artillery = 0.4
	infantry = 0.5
	cavalry = 0.1
	
	morale = 2.0
	
	# Possible handle actions
	handle_action_negotiate = yes
	handle_action_stability = no
	handle_action_build_core = no
	handle_action_send_missionary = no		
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 0
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
		always = yes
	}
	siege_won_effect = {
		owner = {
			if = {
				limit = { has_estate = estate_mages }
				change_estate_land_share = {
					estate = estate_mages
					share = 1.0
					province = ROOT
				}
			}
		}
	}
	
	# Country scope
	can_negotiate_trigger = {
		always = yes
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = "mage_disaster_rebels_demand"
	
	# Country Scope	
	demands_enforced_effect = {
		hidden_effect = {
			add_country_modifier = {
				name = just_lost_to_rebels
				duration = 10
				hidden = yes
			}
		}
		clr_country_flag = has_estate_disaster_flag
		clr_estate_witch_king_points = yes
		
		if = {
			limit = {
				OR = {
					has_government_attribute = locked_T1_reform
					has_government_attribute = locked_government_type
					is_tribal = yes
				}
			}
			set_estate_privilege = estate_mages_puppet_of_corrupt_mages
		}
		else = {
			change_government = theocracy
			add_government_reform = magocracy_reform
		}

		if = {
			limit = { ruler_is_powerful_mage = yes }
			change_adm = -1
			change_dip = -1
			change_mil = -1
		}
		else_if = {
			limit = { heir_is_powerful_mage = yes }
			kill_ruler = yes
			change_adm = -1
			change_dip = -1
			change_mil = -1
		}
		else = {
			kill_heir = { allow_new_heir = no }
			kill_ruler = yes
			define_ruler = {
				change_adm = -1
				change_dip = -1
				change_mil = -1
			}
			custom_tooltip =  will_be_mage_tt
			hidden_effect = { country_event = { id = magic_system.10 } }
		}
		end_disaster = estate_mages_disaster
	}
}
