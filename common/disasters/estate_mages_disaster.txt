
estate_mages_disaster = {
	potential = {
		has_estate = estate_mages
		estate_influence = {
			estate = estate_mages
			influence = 80
		}
		NOT = { has_country_flag = farranean_estates_harmony }
	}

	can_start = {
		has_any_disaster = no
		estate_influence = {
			estate = estate_mages
			influence = 100
		}
		ruler_is_not_witch_king = yes
		NOT = { has_estate_privilege = estate_mages_puppet_of_corrupt_mages }
		custom_trigger_tooltip = {
			tooltip = not_have_mages_cowed_tt
			NOT = { has_estate_influence_modifier = { estate = estate_mages modifier = EST_VAL_MAGES_COWED } }
		}
		is_at_war = no
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			ruler_is_witch_king = yes
			NOT = {
				estate_influence = {
					estate = estate_mages
					influence = 100
				}		
			}
		}
	}
	
	down_progress = {
	}
	
	progress = {
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_mages
					loyalty = 60
				}
			}
		}	
		modifier = {
			factor = 0.5
			custom_trigger_tooltip = {
				tooltip = mages_disaster_respected_tt
				type_has_infamy_respected = { type = country }
				type_has_infamy_paragon = { type = ruler }
			}
		}
		modifier = {
			factor = 1
			custom_trigger_tooltip = {
				tooltip = mages_disaster_suspicious_tt
				type_has_infamy_suspicious = { type = country }
				type_has_infamy_respected_or_better = { type = ruler }
			}
		}
		modifier = {
			factor = 1.5
			custom_trigger_tooltip = {
				tooltip = mages_disaster_denounced_tt
				type_has_infamy_denounced = { type = country }
				type_has_infamy_suspicious_or_better = { type = ruler }
			}
		}
		modifier = {
			factor = 2
			custom_trigger_tooltip = {
				tooltip = mages_disaster_witch_king_tt
				type_has_infamy_witch_king = { type = country }
				ruler_is_not_witch_king = yes
			}
		}
	}
	
	can_end = { NOT = { has_spawned_rebels = mage_disaster_rebels } }
	
	modifier = {	
		mages_mana_regen = -1
		mages_estate_experience_mod = -1
	}

	on_start = estate_disasters.200
	on_end = estate_disasters.201
	
	on_monthly = {
	}
}
