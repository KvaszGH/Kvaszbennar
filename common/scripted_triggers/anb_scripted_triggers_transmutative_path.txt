transmutative_path_owner_has_chi_trigger = {
	variable_arithmetic_trigger = {
		export_to_variable = { which = dev_var value = development }
		export_to_variable = { which = chi_var value = trigger_value:church_power who = owner }
		multiply_variable = { dev_var = 3 }
		check_variable = { which = chi_var which = dev_var }
	}
}

	OR = {
		transmutative_path_is_tamran_and_province_yes = yes
		transmutative_path_is_phonan_and_province_yes = yes
		transmutative_path_is_hinphat_bokai_and_province_yes = yes
		transmutative_path_is_binhrung_and_province_yes = yes
		transmutative_path_is_naamjyut_and_province_yes = yes
		transmutative_path_is_antmvura_and_province_yes = yes
		transmutative_path_is_harimari_and_province_yes = yes
		transmutative_path_is_azkare_and_province_yes = yes
		transmutative_path_is_thirabnir_and_province_yes = yes
		transmutative_path_is_khao_elnak_and_province_yes = yes
		transmutative_path_is_lo_ngoen_and_province_yes = yes
		transmutative_path_is_xia_and_province_yes = yes
		transmutative_path_is_gon_and_province_yes = yes
		AND = { #Generic
			is_capital = yes
			owner = {
				NOT = { primary_culture = antmvura }
				NOT = { culture_group = kai }
				NOT = { culture_group = harimari }
				NOT = { primary_culture = goldscale_kobold }
				NOT = { primary_culture = naamjyut }
				NOT = { has_government_attribute = is_in_xia }
				NOT = { primary_culture = gon }
			}
		}
	}
}

transmutative_path_change_trade_good_trigger = {
	NOT = {
		trade_goods = damestear
		trade_goods = gold
		trade_goods = precursor_relics
	}
	OR = {
		AND = { #Naamjyut
			owner = { has_church_aspect = transmutative_path_tea_ceremonies }
			has_terrain = jungle
			NOT = { trade_goods = tea }
		}
		AND = { #Tamran
			owner = { has_church_aspect = transmutative_path_from_strings_to_silk }
			has_terrain = jungle
			trade_goods = cloth
		}
		AND = { #Phonan
			owner = { has_church_aspect = transmutative_path_dowsing_for_metals }
			OR = {
				has_terrain = hills
				has_terrain = highlands
				has_terrain = mountain
			}
			NOT = { trade_goods = copper }
		}
		AND = { #Hinphat
			owner = { has_church_aspect = transmutative_path_muzzle_the_ox }
			trade_goods = livestock
		}
		AND = { #Binhrung
			owner = { has_church_aspect = transmutative_path_strip_the_scales }
			trade_goods = fish
		}
		AND = { #Antmvura
			owner = { has_church_aspect = transmutative_path_irrigate_jungle_plantations }
			NOT = { trade_goods = coffee }
			has_terrain = jungle
		}
		AND = { #Bokai
			owner = { has_church_aspect = transmutative_path_convert_pasture_lands }
			trade_goods = wool
		}
		AND = { #Azkare
			owner = { has_church_aspect = transmutative_path_weaving_weal }
			trade_goods = cotton
		}
		AND = { #Thirabnir
			owner = { has_church_aspect = transmutative_path_dessicate_landscape }
			OR = {
				has_terrain = drylands
				has_terrain = desert
			}
			NOT = { trade_goods = salt }
		}
		AND = { #Khao Elnak
			owner = { has_church_aspect = transmutative_path_treadeth_the_corn }
			trade_goods = grain
		}
		AND = { #Lo Ngoen
			owner = { has_church_aspect = transmutative_path_dredge_the_flats }
			trade_goods = salt
		}
		AND = { #Gon
			owner = { has_church_aspect = transmutative_path_flavor_of_life }
			has_terrain = jungle
			NOT = { trade_goods = sugar }
		}
	}
}
